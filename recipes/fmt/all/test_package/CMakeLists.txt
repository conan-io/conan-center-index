cmake_minimum_required(VERSION 3.15)
if(FMT_TEST_USE_MODULE)
    cmake_minimum_required(VERSION 3.28)
endif()
project(test_package LANGUAGES CXX)

find_package(fmt REQUIRED CONFIG)

add_executable(test_package test_package.cpp)
if(FMT_TEST_USE_MODULE)
    target_compile_features(test_package PRIVATE cxx_std_20)
    target_compile_definitions(test_package PRIVATE FMT_TEST_USE_MODULE)
    set_target_properties(test_package PROPERTIES CXX_SCAN_FOR_MODULES ON)
else()
    target_compile_features(test_package PRIVATE cxx_std_14)
endif()
if(FMT_HEADER_ONLY)
    target_link_libraries(test_package PRIVATE fmt::fmt-header-only)
else()
    target_link_libraries(test_package PRIVATE fmt::fmt)
endif()

add_executable(test_ranges test_ranges.cpp)
if(FMT_TEST_USE_MODULE)
    target_compile_features(test_ranges PRIVATE cxx_std_20)
    target_compile_definitions(test_ranges PRIVATE FMT_TEST_USE_MODULE)
    set_target_properties(test_ranges PROPERTIES CXX_SCAN_FOR_MODULES ON)
else()
    target_compile_features(test_ranges PRIVATE cxx_std_14)
endif()
if(FMT_HEADER_ONLY)
    target_link_libraries(test_ranges PRIVATE fmt::fmt-header-only)
else()
    target_link_libraries(test_ranges PRIVATE fmt::fmt)
endif()
