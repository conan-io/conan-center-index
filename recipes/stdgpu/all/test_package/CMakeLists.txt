cmake_minimum_required(VERSION 3.15)
project(test_package CXX)

find_package(stdgpu CONFIG REQUIRED)

if (stdgpu_VERSION_STRING EQUAL "1.3.0")
    set(CMAKE_CXX_STANDARD 14)
else()
    set(CMAKE_CXX_STANDARD 17)
endif()

if (STDGPU_BACKEND STREQUAL "cuda")
    cmake_minimum_required(VERSION 3.18)
    enable_language(CUDA)
    find_package(CUDAToolkit REQUIRED)
    add_executable(${PROJECT_NAME} test_package_cuda.cu)
    target_link_libraries(${PROJECT_NAME} CUDA::cudart stdgpu::stdgpu)
    set_target_properties(${PROJECT_NAME} PROPERTIES CUDA_ARCHITECTURES OFF)
elseif(STDGPU_BACKEND STREQUAL "openmp")
    add_executable(${PROJECT_NAME} test_package_omp.cpp)
    target_link_libraries(${PROJECT_NAME} stdgpu::stdgpu)
endif()
